  public int minValue(int N, int[] s)
  {
    int[] buyItemCnt = new int[N];
    int M = s.length;
    int i, j, k, ret=0;
    
    for (i = 0; i < N; i++) {
      buyItemCnt[i] = 0;
    }
    
    for (j=0; j < M; j++) {
      while (s[j] > 0) {
        int minCntIndex = 0;
        for (i = 1; i < N; i++) {
          if (buyItemCnt[minCntIndex] > buyItemCnt[i]) minCntIndex = i;
        }
        
        buyItemCnt[minCntIndex]++;
        s[j]--;
      }
    }
    
    for (i = 0; i < N; i++) {
      if (buyItemCnt[i] == M) ret++;
    }
    
    return ret;
  }
